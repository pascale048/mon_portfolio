Sub Archiver_Produits_Avec_Date()

    Dim wsSrc As Worksheet
    Dim wsDst As Worksheet
    Dim col As Long
    Dim lastRow As Long
    Dim i As Long
    Dim dataArr As Variant
    Dim cell As Range
    Dim tempArr() As Variant

    Set wsSrc = Worksheets("Feuille1")
    Set wsDst = Worksheets("archive")

    ' --- ARCHIVAGE PAR PRODUIT (1 colonne = 1 ligne) ---
    For col = 2 To 7   ' B à G

        ' Vérifie si le produit existe (ligne 4)
        If Trim(wsSrc.Cells(4, col).Value) <> "" Then

            ' Trouve la première ligne vide dans archive
            lastRow = wsDst.Cells(wsDst.Rows.Count, "A").End(xlUp).Row + 1

            ' Date du jour
            wsDst.Cells(lastRow, 1).Value = Date

            ' Récupère toutes les valeurs de la colonne (lignes 4 à 91)
            dataArr = wsSrc.Range(wsSrc.Cells(4, col), wsSrc.Cells(91, col)).Value

            ' On crée un tableau temporaire contenant uniquement les VALEURS
            ReDim tempArr(1 To UBound(dataArr, 1), 1 To 1)
            For i = 1 To UBound(dataArr, 1)
                tempArr(i, 1) = dataArr(i, 1)   ' copie valeur brute
            Next i

            ' Copie transposée en ligne
            wsDst.Cells(lastRow, 2).Resize(1, UBound(tempArr, 1)).Value = _
                Application.WorksheetFunction.Transpose(tempArr)

        End If

    Next col

    ' --- NETTOYAGE DES DONNÉES SUR FEUILLE1 ---
    For Each cell In wsSrc.Range("B4:G91")
        If Not cell.HasFormula Then
            cell.ClearContents   ' efface uniquement les valeurs
        End If
    Next cell

    MsgBox "Archivage terminé. Les valeurs ont été copiées, les formules et listes conservées.", vbInformation

End Sub
